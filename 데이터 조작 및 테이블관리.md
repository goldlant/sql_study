# 데이터 조작 및 테이블 관리

__INSERT 문법__
<pre>
<code>
--테이블이 만들어지면 빈 공간이 만들어 지는 것이며 테이블 안에 데이터를 INSERT하는 것이 필요하다.
</code>
</pre>
<pre>
<code>
INSERT 
INTO
	TBALE_NAME		--INSRT할 테이블 지정
VALUES
(
	VALUES 1,		--각 컬럼 값을 입력한다.
	VALUES 2,
	VALUES 3,
)
</code>
</pre>
--------------------------------------------------------------------------------------------------------------------
<pre>
<code>
INSERT
INTO
	TABLE_NAME(COLUMN1, COLUMN2)
VALUES
(
	VALUES 1,		--각 컬럼 값을 입력한다.
	VALUES 2,
	VALUES 3,
);
</code>
</pre>

__UPDATE 문법__
<pre>
<code>
--테이블의 존재하는 데이터를 ㅅ정하는 작업, 업무를 처리하는데 필수적인 것이며 동시성에 유의해야 한다.
  </code>
</pre>
<pre>
<code>
UPDATE
	TABLE_NAME				--UPDATE할 테이블 지정
SET							--수정할 컬럼 및 값 입력
	COLUMN_1=VALUE1,
	COLUMN_2=VALUE2
WHERE 조건;					--대상 조건
  </code>
</pre>

1. UPDATE는 대상 행에 대해서 락을 잡는다.
2. 락이란 다른 사용자는 해당 행에 대해서 작업을 하지 못한다는 것이다.
3. 즉 UPDATE를 한 후 재빨리 COMMIT을 하지 않는다면 RDBMS의 동시성이 낮아진다.
--------------------------------------------------------------------------------------------------------------------
__테이블 준비__
<pre>
<code>
ALTER TABLE LINK ADD COLUMN LAST_UPDATE DATE;		--LINK 테이블에 LAST_UPDATE 컬럼 추가
ALTER TABLE LINK ALTER COLUMN LAST_UPDATE SET DEFAULT CURRENT_DATE;	--LAST_UPDATE 컬럼의 기본값을 현재시간으로 함
  </code>
</pre>
--------------------------------------------------------------------------------------------------------------------
<pre>
<code>
UPDATE LINK 
SET 
	LAST_UPDATE = DEFAULT 		--LAST_UPDATE컬럼의 값을 지정한 DEFAULT값으로 UPDATE한다.
WHERE 
	LAST_UPDATE IS null			--LAST_UPDATE컬럼이 NULL인 값을 대상으로 한다.
COMMIT;
  </code>
</pre>
-------------------------------------------------------------------------------------------------------------------
<pre>
<code>
UPDATE LINK 
SET
	REL = 'NO DATA';		--REL컬럼의 값을 'NO DATA'로 수정한다. WHERE 절이 없으므로 테이블의 전체 행을 대상으로 한다.
COMMIT;
  </code>
</pre>
-------------------------------------------------------------------------------------------------------------------
<pre>
<code>
UPDATE LINK 
SET
	DESCRIPTION = NAME;		--DESCRIPTION컬럼의 값을 NAME컬럼으로 수정한다. WHERE 절이 없으므로 테이블의 전체 행을 대상으로 한다.
COMMIT;
  </code>
</pre>

__UPDATE JOIN 문법__
<pre>
<code>
-- UPDATE시 다른 테이블의 내용을 참조 하고 싶을 때 UPDATE JOIN문을 사용한다. 복잡한 업무를 처리하는데 매우 유용한 방법이다.
  </code>
</pre>
<pre>
<code>
UPDATE
	TARGET_TABLE A		--UPDATE할 테이블 지정
SET 					
	A.COLUMN_1 = 표현식	--특정 컬럼 UPDATE
FROM 
	REF_TABLE B			--참조 테이블 지정
WHERE A.COLUMN_1 = B.COLUMN_1;	--조인조건
  </code>
</pre>

__DELETE 문법__
<pre>
<code>
--DELETE문은 테이블에서 특정 데이터를 삭제하거나 테이블 내에 존재하는 모든 데이터를 삭제할 수 있다.
  </code>
</pre>

<pre>
<code>
DELETE 
FROM 					--삭제할 테이블
	TARGET_TABLE A
WHERE 조건식				--삭제할 행에 대한 조건
;
  </code>
</pre>
-----------------------------------------------------------------------------------------------------------
<pre>
<code>
DELETE 
FROM 
	LINK 
WHERE ID=5;		--ID가 5인 행을 삭제
  </code>
</pre>
-----------------------------------------------------------------------------------------------------------
<pre>
<code>
DELETE
FROM 
	LINK_TEMP A		--LINK_TMP 테이블을 삭제
USING LINK B		--LINK 테이블과 조인
WHERE A.ID=B.ID		--ID 컬럼 기준
  </code>
</pre>
-----------------------------------------------------------------------------------------------------------
<pre>
<code>
DELETE FROM LINK;		--LINK 테이블 삭제

DELETE FROM LINK_TMP;	--LINK_TMP 테이블 삭제
  </code>
</pre>

__UPSERT 문법__
<pre>
<code>
__INSERT를 시도할때 조건(상황)에 따라 UPDATE를 할수 있는 구문이다. 복잡한 업무 처리에 자주 사용 된다.
  </code>
</pre>

<pre>
<code>
INSERT 
INTO
	TABLE_NAME(COLUMN_1)
VALUES(VALUE_1)		--INSERT 시도
ON CONFLICT TARGET ACTION;		--충돌 시 다른 액션
  </code>
</pre>

__테이블 준비__
<pre>
<code>
CREATE TABLE CUSTOMERS
(
	CUSTOMER_ID SERIAL PRIMARY KEY,
	NAME VARCHAR UNIQUE,
	EMAIL VARCHAR NOT NULL,
	ACTIVE BOOL NOT NULL DEFAULT TRUE
);

INSERT 
INTO CUSTOMERS(NAME,EMAIL)
VALUES
	('IBM','CONTACT@IBL.COM'),
	('MICROSOFT','CONTACT@MICROSOFT.COM'),
	('INTEL','CONTACT@INTEL.COM')
	  </code>
</pre>
-----------------------------------------------------------------------------------------------------------
<pre>
<code>
INSERT 
INTO 
	CUSTOMERS(NAME,EMAIL)		--CUSTOMERS 테이블에 INSERT 시도
VALUES
	('MICROSOFT','HOTLINE@MICROSOFT.COM')	--기존에 존재하는 NAME
ON CONFLICT (NAME)		--충돌 시(기존에 존재할 경우)
DO NOTHING 				--아무것도 안함
	  </code>
</pre>
-----------------------------------------------------------------------------------------------------------
<pre>
<code>
INSERT 
INTO 
	CUSTOMERS(NAME,EMAIL)		--CUSTOMERS 테이블에 INSERT 시도
VALUES
	('MICROSOFT','HOTLINE@MICROSOFT.COM')	--기존에 존재하는 NAME
ON CONFLICT (NAME)		--충돌 시(기존에 존재할 경우)
DO UPDATE 
		SET EMAIL = EXCLUDED.EMAIL ||';'||CUSTOMERS.EMAIL;	--EXCLUDED.EMAIL은 위에서 INSERT시도한 EMAIL값을 가리킴
			 </code>
</pre>

__EXPORT 작업__
<pre>
<code>
--EXPORT는 테이블의 데이터를 다른 형태의 데이터로 추출 하는 작업이다. 대표적으로 CSV형식으로 가장 많이 추출한다.
</code>
</pre>
<pre>
<code>
COPY CATEGORY(CATEGORY_ID, NAME, LAST_UPDATE)	--추출할 테이블과 컬럼을 지정한다.
TO 'C:\TMP\DB_CATEGORY.CSV'			--추출할 데이터를 저장할 파일을 지정한다.
DELIMITER ','						--구분자를 지정한다.
CSV HEADER;							--파일형식을 지정한다.
 </code>
</pre>
------------------------------------------------------------------------------------------------------------
<pre>
<code>
COPY CATEGORY(CATEGORY_ID, NAME, LAST_UPDATE)	--추출할 테이블과 컬럼을 지정한다.

 </code>

);
 </code>
