__UNION 연산 문법__
<pre>
<code>
--두 개 이상의 SELECT 문들의 결과 집합을 단일 결과 집합으로 결합하며 결합시 중복된 데이터는 제거 된다.
</code>
</pre>
<pre>
<code>
SELECT 
	COLUMN_1_1,				--두개의 SELECT문 간 컬럼의 개수는 동일해야 하고 해당 순서의 열에는 서로 호환되는 데이터 유형이어야 한다. 
	COLUMN_2_2
FROM 
	TABLE_NAME_1
UNION
SELECT
	COLUMN_2_1,				
	COLUMN_2_2
FROM 
	TABLE_NAME_2;
  </code>
</pre>

__서브쿼리란?__
<pre>
<code>
--서브쿼리는 SQL문 내에서 메인 쿼리가 아닌 하위에 존재하는 쿼리를 말한다. 서브쿼리를 활용함으로써 다양한 결과를 도출 할 수 있다.
  </code>
</pre>
<pre>
<code>
SELECT 
	FILM_ID,
	TITLE,
	RENTAL_RATE
FROM FILM 
WHERE RENTAL_RATE >			--중첩 서브쿼리의 시작
(
	SELECT 
		AVG(RENTAL_RATE)
	FROM 
		FILM
);							--중첨 서브쿼리의 종료
  </code>
</pre>
---------------------------------------------------------------------------------------------------------------
__인라인 뷰의 활용__
<pre>
<code>
SELECT 
	A.FILM_ID,
	A.TITLE,
	A.RENTAL_RATE
FROM 
	FILM A,
	(SELECT			--인라인 뷰 시작
			AVG(RENTAL_RATE) AS AVG_RENTAL_RATE
	 FROM 
	 		FILM 
	 )B
WHERE A.RENTAL_RATE > B.AVG_RENTAL_RATE		--인라인 뷰 종료
 </code>
</pre>
--------------------------------------------------------------------------------------------------------------
__스칼라 서브쿼리의 활용__
<pre>
<code>
SELECT 
	A.FILM_ID,
	A.TITLE,
	A.RENTAL_RATE 
FROM 
(
SELECT
	A.FILM_ID,
	A.TITLE,
	A.RENTAL_RATE,
	(										--스칼라 서브쿼리의 시작
		SELECT AVG(L.RENTAL_RATE)
		FROM FILM L
	)AS AVG_RENTAL_RATE						--스칼라 서브쿼리의 종료
FROM FILM A
) A											--인라인 뷰의 종료
WHERE A.RENTAL_RATE > A.AVG_RENTAL_RATE;
 </code>
</pre>

__ANY 연산자 문법__
<pre>
<code>
-- ANY 연산자는 값을 서브 쿼리에 의해 반환된 값 집합과 비교한다. ANY 연산자는 서브쿼리의 값이 어떠한 값이라도 만족을 하면 조건이 성립된다.
 </code>
</pre>
  
<pre>
<code>
SELECT TITLE, LENGTH
FROM FILM 
WHERE LENGTH >=ANY					--영화분류별 상영시간이 가장 긴 영화의 제목 및 상영시간을 구함, any가 없다면 sql에러가 발생 -> 이 경우 서브쿼리의 반환되는 집합은 단 한건이어야만 함
(
	SELECT MAX(LENGTH)
	FROM FILM A, FILM_CATEGORY B
	WHERE A.FILM_ID = B.FILM_ID 
	GROUP BY B.CATEGORY_ID
);
 </code>
</pre>
----------------------------------------------------------------------------------------------------------------
<pre>
<code>
SELECT TITLE, LENGTH
FROM FILM 
WHERE LENGTH = ANY		-- '=ANY' = 'IN'
(						--영화분류별 상영시간이 가장 긴 영화의 제목 및 상영시간을 구함, ANY가 없다면 sql에러가 발생 -> 이 경우 서브쿼리의 반환되는 집합은 단 한건이어야만 함
	SELECT MAX(LENGTH)
	FROM FILM A, FILM_CATEGORY B
	WHERE A.FILM_ID = B.FILM_ID 
	GROUP BY B.CATEGORY_ID
);
 </code>
</pre>

__ALL 연산자 문법__
<pre>
<code>
-- 서브 쿼리에 의해 반환된 값 집합과 비교한다. ALL 연산자는 서브쿼리의 모든 값이 만족을 해야만 조건이 성립
 </code>
</pre>
<pre>
<code>
SELECT TITLE, LENGTH
FROM FILM 
WHERE LENGTH >=ALL				--영화분류별 상영시간이 가장 긴 영화의 제목 및 상영시간을 구함, ALL이 없다면 sql에러가 발생 -> 이 경우 서브쿼리의 반환되는 집합은 단 한건이어야만 함
(
	SELECT MAX(LENGTH)
	FROM FILM A, FILM_CATEGORY B
	WHERE A.FILM_ID = B.FILM_ID 
	GROUP BY B.CATEGORY_ID
);
 </code>
</pre>
-------------------------------------------------------------------------------------------------------------------
<pre>
<code>
SELECT
	FILM_ID,
	TITLE,
	LENGTH
FROM FILM 
WHERE LENGTH > ALL
(
	SELECT ROUND(AVG(LENGTH),2)		-- 평가기준 평균값들보다 상영시간이 긴 영화정보 출력
	FROM FILM 
	GROUP BY RATING

)
ORDER BY LENGTH
 </code>

__EXISTS 연산자__
<pre>
<code>
--EXISTS 연산자는 서브쿼리 내에 집합이 존재하는지 존재 여부만을 판단, 존재 여부만을 판단하므로 연산 시 부하가 줄어듬
--해당 집합이 존재하기만 하면 더이상 연산을 멈추므로 성능상 유리함
 </code>
</pre>
<pre>
<code>
SELECT 
	FIRST_NAME,
	LAST_NAME
FROM 
	CUSTOMER C
WHERE
EXISTS 
( 
	SELECT 1
	FROM PAYMENT P
	WHERE P.CUSTOMER_ID = C.CUSTOMER_ID 
	AND P.AMOUNT >11			
)
ORDER BY FIRST_NAME, LAST_NAME;
 </code>
</pre>
--------------------------------------------------------------------------------------------------------------------
<pre>
<code>
SELECT 
	FIRST_NAME,
	LAST_NAME
FROM 
	CUSTOMER C
WHERE
NOT EXISTS 
( 
	SELECT 1
	FROM PAYMENT P
	WHERE P.CUSTOMER_ID = C.CUSTOMER_ID 
	AND P.AMOUNT >11			
)
ORDER BY FIRST_NAME, LAST_NAME;
 </code>
</pre>
