# 테이블 관리


__데이터 타입__
<pre>
<code>
--테이블은 컬럼으로 이루어져 있고 컬럼은 다양한 데이터 타입을 지원한다. 이는 RDBMS가 제 역할을 하는데 있어서 매우 중요하다.
</code>
</pre>
__다양한 데이터 타입 지원 - BOOLEAN, CHARACTER, NUMERIC__


<img width="1222" alt="KakaoTalk_20210614_171619470" src="https://user-images.githubusercontent.com/72371379/121861113-83503f80-cd34-11eb-90d0-ae23ba399072.png">

__테이블 준비__
<pre>
<code>
CREATE TABLE DATA_TYPE_TEST_1
(
	A_BOOLEAN BOOLEAN,
	B_CHAR CHAR(10),
	C_VARCHAR VARCHAR(10),
	D_TEXT TEXT,
	E_INT TINT,
	F_SMALLINT SMALLINT,
	G_FLOAT FLOAT,
	H_NUMERIC(15,2)
);

INSERT 
INTO
	DATA_TYPE_TEST_1
VALUES
(
	 TRUE, --A_BOOLEAN
	'ABCDE'. --B_CHAR 
	'ABCDE', --C_VARCHAR
	'TEXT', --D_TEXT
	 1000, --E_INT
	 10, --F_SMALLINT
	 10,12345, --G_FLOAT 
	 10.25 --H_NUMERIC
);
</code>
</pre>
<pre>
<code>
SELECT * FROM DATA_TYPE_TEST_1;
</code>
</pre>

__다양한 데이터 타입 지원 - TIME, ARRAYS, JSON__
<img width="1223" alt="KakaoTalk_20210614_171709932" src="https://user-images.githubusercontent.com/72371379/121861293-ad096680-cd34-11eb-938b-4337009efdd1.png">

__테이블 준비__
<pre>
<code>
CREATE TABLE DATA_TYPE_TEST_2
(
	A_DATE DATE,
	B_TIME TIME,
	C_TIMESTAMP TIMESTAMP,
	D_ARRAY TEXT[],
	E_JSON JSON
);

INSERT 
INTO
	DATA_TYPE_TEST_2
VALUES
(
	CURRENT_DATE, --A_DATE
	LOCALTIME, --B_TIME
	CURRENT_TIMESTAMP, --C_TIMESTAMP
	ARRAY['010-1234-1234','010-4321-4321'],
	'{"CUSTOMER":"JOHN DOE","ITEMS":{"PRODUCT":"BEER","QTY":6}}' --E_JSON
);
</code>
</pre>
<pre>
<code>
SELECT * FROM DATA_TYPE_TEST_2
</code>
</pre>
