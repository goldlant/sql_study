# 데이터 조작 및 테이블 관리

__INSERT 문법__
<pre>
<code>
--테이블이 만들어지면 빈 공간이 만들어 지는 것이며 테이블 안에 데이터를 INSERT하는 것이 필요하다.
</code>
</pre>
<pre>
<code>
INSERT 
INTO
	TBALE_NAME		--INSRT할 테이블 지정
VALUES
(
	VALUES 1,		--각 컬럼 값을 입력한다.
	VALUES 2,
	VALUES 3,
)
</code>
</pre>

<code>

	TABLE_NAME(COLUMN1, COLUMN2)
VALUES

	VALUES 1,		--각 컬럼 값을 입력한다.
	VALUES 2,
	VALUES 3,
);
</code>
</pre>

__테이블 준비__
<pre>
<code>
CREATE TABLE LINK(
	ID SERIAL PRIMARY KEY,
	URL VARCHAR(255) NOT NULL,
	NAME VARCHAR(255) NOT NULL,
	DESCRIPTION VARCHAR(255),
	REL VARCHAR(50)
);
</code>
</pre>

<code>
INSERT
INTO
	LINK(URL,NAME)		--Link 테이블에 insert 컬럼 지정(컬럼 지정을 하지 않으면 자동으로 NULL이 들어감)
VALUES ('http://naver.com','Naver');	--값 입력
</code>

<pre>
<code>
CREATE TABLE LINK_TEMP AS		--LINK테이블의 스키마만 가져와서 LINK_TMP 테이블을 생성한다.
SELECT * FROM LINK WHERE 0=1	--LINK_TMP 테이블 구조는 LINK와 같고 데이터는 0건이 된다.

INSERT 
INTO LINK_TEMP
SELECT * 
FROM LINK;

SELECT * FROM LINK_TEMP A
EXCEPT                --A-B=0
SELECT * FROM LINK B

SELECT * FROM LINK B
EXCEPT                --B-A=0
SELECT * FROM LINK_TEMP A
  </code>

__UPDATE 문법__
<pre>
<code>
--테이블의 존재하는 데이터를 ㅅ정하는 작업, 업무를 처리하는데 필수적인 것이며 동시성에 유의해야 한다.
  </code>
</pre>
<pre>
<code>
UPDATE
	TABLE_NAME				--UPDATE할 테이블 지정
SET							--수정할 컬럼 및 값 입력
	COLUMN_1=VALUE1,
	COLUMN_2=VALUE2
WHERE 조건;					--대상 조건
  </code>
</pre>

1. UPDATE는 대상 행에 대해서 락을 잡는다.
2. 락이란 다른 사용자는 해당 행에 대해서 작업을 하지 못한다는 것이다.
3. 즉 UPDATE를 한 후 재빨리 COMMIT을 하지 않는다면 RDBMS의 동시성이 낮아진다.
--------------------------------------------------------------------------------------------------------------------
__테이블 준비__
<pre>
<code>
ALTER TABLE LINK ADD COLUMN LAST_UPDATE DATE;		--LINK 테이블에 LAST_UPDATE 컬럼 추가
ALTER TABLE LINK ALTER COLUMN LAST_UPDATE SET DEFAULT CURRENT_DATE;	--LAST_UPDATE 컬럼의 기본값을 현재시간으로 함
  </code>
</pre>
--------------------------------------------------------------------------------------------------------------------
<pre>
<code>
UPDATE LINK 
SET 
	LAST_UPDATE = DEFAULT 		--LAST_UPDATE컬럼의 값을 지정한 DEFAULT값으로 UPDATE한다.
WHERE 
	LAST_UPDATE IS null			--LAST_UPDATE컬럼이 NULL인 값을 대상으로 한다.
COMMIT;
  </code>
</pre>
-------------------------------------------------------------------------------------------------------------------
<pre>
<code>
UPDATE LINK 
SET
	REL = 'NO DATA';		--REL컬럼의 값을 'NO DATA'로 수정한다. WHERE 절이 없으므로 테이블의 전체 행을 대상으로 한다.
COMMIT;
  </code>
</pre>
-------------------------------------------------------------------------------------------------------------------
<pre>
<code>
UPDATE LINK 
SET
	DESCRIPTION = NAME;		--DESCRIPTION컬럼의 값을 NAME컬럼으로 수정한다. WHERE 절이 없으므로 테이블의 전체 행을 대상으로 한다.
COMMIT;
  </code>
</pre>
